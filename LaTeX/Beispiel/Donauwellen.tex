\begin{recipe}
[ % Optionale Eingaben
    bakingtime={\unit[30]{min}},
    bakingtemperature={\unit[200]{\textcelcius}},
%    portion = {\portion{5-6}}
]
{Donauwellen}
    
    \graph
    {% Bilder
%        small=glass,    % kleines Bild
        big=./Bilder/Donauwellen % großes (längeres) Bild
    }
    
    \ingredients
    {% Zutaten
	    \unit[250]{g} & Butter \\
		\unit[350]{g} & Mehl \\
		6 & Eier\\
		\unit[200]{g} & Zucker\\
		\unit[1]{Pck.} & Vanillezucker \\
		\unit[1/2]{Pck.} & Backpulver \\
		2 Gläser & Sauerkirschen\\
		\unit[2]{EL} & Kakao\\
		\ \\
		\unit[1]{Pck.} & Paradiescreme Vanille \\
		\unit[400]{ml} & Schlagsahne \\
		\ \\
		\unit[1-2]{Pck.}& Schokoglasur \\
    }
    
        
    
    
    \preparation
    { % Zubereitung
        \step Butter schaumig rühren. Nach und Nach Zucker und Eier zugeben. Mehl und Backpulver mischen und esslöffelweise zugeben.
        \step Etwa die Hälfte des Teiges auf das Backblech streichen. Den restlichen Teig mit dem Kakao verrühren und auf dem hellen Teig verteilen.
        \step Mit einer Gabel Wellen im Teig ziehen. Die abgetropften Kirschen auf dem Teig verteilen. Bei \unit[200]{\textcelcius} für \unit[30]{min} backen.
        \step Die Paradiescreme mit der Sahne für ungefähr \unit[3]{min} mit dem Handrührgerät schlagen und auf dem erkalteten Kuchen verteilen. Mit der Schokoglasur verzieren.
    }
    
    \hint
    {% Tipp
        Eine Portion Paradiescreme ist wenig, zwei sind viel.
    }

\end{recipe}
